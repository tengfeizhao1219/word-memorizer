## 🔧 技术实现

```javascript
// 模拟登录函数
function useMockLogin(userInfo) {
  // 生成模拟数据
  // 保存到本地存储
  // 跳转到首页
}

// 模拟数据加载
function loadMockData() {
  // 从本地存储加载
  // 更新页面数据
}
```

### 自动切换:
```javascript
// 云函数调用失败时自动切换
fail: (err) => {
  console.error('云函数失败:', err);
  // 自动调用 useMockLogin
  useMockLogin(userInfo);
}
```
